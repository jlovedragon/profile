*** git 
   + $ mkdir your_repo
   + $ cd your_repo
   + $ git init
   + $ git pull git@github.com:yourname/yourrepo.git
   + $ git add -u
   + $ git remote add origin git@github.com:yourname/yourrepo.git
   + $ git commit -m ""
   + $ git push origin master
